<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-semibold dark:text-white text-gray-900 mb-3">Quick Start</h2>
  <p className="dark:text-gray-400 text-gray-600 mb-8">Get up and running with TopK in minutes</p>
  
  <Steps>
    <Step title="Install the SDK">
      Choose your preferred language and install the TopK SDK:
      
      <CodeGroup>
      ```bash npm
      npm install topk-js
      ```
      ```bash yarn
      yarn add topk-js
      ```
      ```bash pip
      pip install topk-sdk
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Get your API key">
      Sign up at [console.topk.io](https://console.topk.io) and generate your API key. Initialize the client:
      
      <Tabs>
        <Tab title="JavaScript">
        ```javascript
        import { Client } from "topk-js"
        
        const client = new Client({
          apiKey: "YOUR_TOPK_API_KEY",
          region: "aws-us-east-1-elastica"
        });
        ```
        </Tab>
        <Tab title="Python">
        ```python
        from topk_sdk import Client
        
        client = Client(
          api_key="YOUR_TOPK_API_KEY",
          region="aws-us-east-1-elastica"
        )
        ```
        </Tab>
      </Tabs>
    </Step>
    
    <Step title="Create your first collection">
      Define a schema and create a collection with semantic search enabled:
      
      <Tabs>
        <Tab title="JavaScript">
        ```javascript
        import { text, semanticIndex } from "topk-js/schema"
        
        await client.collections().create("books", {
          title: text().required().index(semanticIndex()),
        });
        ```
        </Tab>
        <Tab title="Python">
        ```python
        from topk_sdk.schema import text, semantic_index
        
        client.collections().create(
          "books",
          schema={
            "title": text().required().index(semantic_index()),
          }
        )
        ```
        </Tab>
      </Tabs>
    </Step>
    
    <Step title="Run your first search">
      Add documents and perform a semantic search:
      
      <Tabs>
        <Tab title="JavaScript">
        ```javascript
        import { select, fn, field } from "topk-js/query"
        
        // Add documents
        await client.collection("books").upsert([
          { _id: "gatsby", title: "The Great Gatsby" },
          { _id: "1984", title: "1984" },
        ]);
        
        // Search with semantic similarity
        const results = await client.collection("books").query(
          select({
            title: field("title"),
            similarity: fn.semanticSimilarity("title", "classic American novel"),
          }).topk(field("similarity"), 10)
        );
        ```
        </Tab>
        <Tab title="Python">
        ```python
        from topk_sdk.query import select, field, fn
        
        # Add documents
        client.collection("books").upsert([
          {"_id": "gatsby", "title": "The Great Gatsby"},
          {"_id": "1984", "title": "1984"},
        ])
        
        # Search with semantic similarity
        results = client.collection("books").query(
          select(
            "title",
            similarity=fn.semantic_similarity("title", "classic American novel")
          ).topk(field("similarity"), 10)
        )
        ```
        </Tab>
      </Tabs>
    </Step>
  </Steps>
</div>
